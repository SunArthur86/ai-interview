# 手撕多头注意力机制(用Pytorch)(Multi Head Attention)

多头注意力机制(Multi-Head Attention)是 Transformer 模型中的核心组件,用于捕捉不同位置之 间的依赖关系.通过将注意力计算分成多个头部,通过多个头,模型可以在不同的投影空间中计算注 意力,最终将各个头的输出拼接,并通过线性层进行整合.具体过程为:

MultiHead(Q, K, V ) = Concat(head1, ... , headh)W

O import torch import torch.nn as nn import torch.nn.functional as F import math class MultiHeadAttention(nn.Module): def __init__(self, heads, d_model, dropout=0.1): super().__init__()
# 保存输入的d_model(模型维度)和头的数量
self.d_model = d_model
# d_k是每个头的特征维度,通常是d_model / heads
self.d_k = d_model // heads self.h = heads
# 三个线性变换层,分别用于将输入映射为Q、K、V
self.q_linear = nn.Linear(d_model, d_model) # 线性层,映射输入到Q self.v_linear = nn.Linear(d_model, d_model) # 线性层,映射输入到V self.k_linear = nn.Linear(d_model, d_model) # 线性层,映射输入到K
# Dropout层,用于防止过拟合
self.dropout = nn.Dropout(dropout)
# 输出的线性层,最终将多头输出的结果合并回原始d_model维度
self.out = nn.Linear(d_model, d_model) def attention(self, q, k, v, d_k, mask=None, dropout=None):
# 1. 计算注意力分数(QKᵀ / √d_k)
# 其中q的维度是 (batch_size, heads, seq_len, d_k)
# k的维度是 (batch_size, heads, seq_len, d_k)
# 计算的结果是 (batch_size, heads, seq_len_q, seq_len_k)
scores = torch.matmul(q, k.transpose(-2, -1)) / math.√d_k

scores = F.softmax(scores, dim=-1)

# 4. 如果有dropout层,应用于注意力权重上
if dropout is not None: scores = dropout(scores)

# 5. 使用注意力权重对V矩阵进行加权求和
# q和k的相似性决定了v的加权方式,输出的维度是 (batch_size, heads,
seq_len_q, d_k) output = torch.matmul(scores, v) return output def forward(self, q, k, v, mask=None): bs = q.size(0) # 获取batch_size大小

# 1. 对Q、K、V分别进行线性变换
# 输入q, k, v的维度为 (batch_size, seq_len, d_model)
# 通过线性层后,维度依旧为 (batch_size, seq_len, d_model)
k = self.k_linear(k) # 线性变换后的k q = self.q_linear(q) # 线性变换后的q v = self.v_linear(v) # 线性变换后的v
# 2. 维度变换:将d_model拆分为多个头 (heads)
# 将Q、K、V重新reshape为 (batch_size, seq_len, heads, d_k)
# 然后再交换维度,得到 (batch_size, heads, seq_len, d_k)
k = k.view(bs, -1, self.h, self.d_k).transpose(1, 2) q = q.view(bs, -1, self.h, self.d_k).transpose(1, 2) v = v.view(bs, -1, self.h, self.d_k).transpose(1, 2)
# 3. 计算注意力,返回的 scores 维度为 (batch_size, heads, seq_len_q, d_k)
scores = self.attention(q, k, v, self.d_k, mask, self.dropout)
# 4. 把多头的输出拼接起来
# 先将 (batch_size, heads, seq_len, d_k) 变回 (batch_size, seq_len,
heads, d_k)

concat = scores.transpose(1, 2).contiguous().view(bs, -1, self.d_model)
