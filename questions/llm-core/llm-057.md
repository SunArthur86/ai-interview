# 手撕多头注意力机制(用Pytorch)(Multi Head Attention)

```python
import torch
import torch.nn as nn
import torch.nn.functional as F
import math

class MultiHeadAttention(nn.Module):
    def __init__(self, heads, d_model, dropout=0.1):
        super().__init__()
        assert d_model % heads == 0, "d_model must be divisible by heads"
        self.d_model = d_model
        self.d_k = d_model // heads
        self.h = heads
        
        # 定义线性变换层
        self.q_linear = nn.Linear(d_model, d_model)
        self.v_linear = nn.Linear(d_model, d_model)
        self.k_linear = nn.Linear(d_model, d_model)
        
        self.dropout = nn.Dropout(dropout)
        self.out = nn.Linear(d_model, d_model)

    def attention(self, q, k, v, d_k, mask=None, dropout=None):
        scores = torch.matmul(q, k.transpose(-2, -1)) / math.sqrt(d_k)
        if mask is not None:
            scores = scores.masked_fill(mask == 0, -1e9)
        scores = F.softmax(scores, dim=-1)
        if dropout is not None:
            scores = dropout(scores)
        return torch.matmul(scores, v)

    def forward(self, q, k, v, mask=None):
        bs = q.size(0)
        
        # 1. 线性变换并分割成多头
        # view 后形状: (bs, seq_len, h, d_k)
        # transpose 后形状: (bs, h, seq_len, d_k) -> 为了让每个头独立计算注意力
        q = self.q_linear(q).view(bs, -1, self.h, self.d_k).transpose(1, 2)
        k = self.k_linear(k).view(bs, -1, self.h, self.d_k).transpose(1, 2)
        v = self.v_linear(v).view(bs, -1, self.h, self.d_k).transpose(1, 2)
        
        # 2. 计算注意力
        scores = self.attention(q, k, v, self.d_k, mask, self.dropout)
        
        # 3. 拼接多头
        # transpose 回: (bs, seq_len, h, d_k)
        # contiguous 确保内存连续性以便 view
        # view 压缩最后两维: (bs, seq_len, d_model)
        concat = scores.transpose(1, 2).contiguous().view(bs, -1, self.d_model)
        
        # 4. 输出线性变换
        output = self.out(concat)
        return output
```

**## 常见考点**
*   **为什么要多头**：多头注意力机制相比单头有什么优势？它捕捉了哪些不同特征子空间？
*   **维度限制**：为什么 $d_{model}$ 必须能被 $h$ (头数) 整除？
*   **Mask 在哪里生效**：Mask 是在每个头独立生效还是共享？Mask 的形状需要注意什么？
