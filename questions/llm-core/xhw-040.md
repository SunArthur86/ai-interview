import torch
import torch.nn as nn
class FeedForwardNetwork(nn.Module):
    def __init__(self, input_dim, hidden_dim, output_dim):
        """
        初始化前馈神经网络。
        参数：
        - input_dim: 输入维度
        - hidden_dim: 隐藏层的维度
        - output_dim: 输出维度，通常与输入维度相同
        """
        super(FeedForwardNetwork, self).__init__()

        # 第一个线性层，将输入维度映射到隐藏维度
        self.linear1 = nn.Linear(input_dim, hidden_dim)
        # 激活函数，使用ReLU非线性激活
        self.relu = nn.ReLU()
        # 第二个线性层，将隐藏层维度映射回输出维度
        self.linear2 = nn.Linear(hidden_dim, output_dim)
    def forward(self, x):
        """
        前向传播函数，定义如何将输入 x 通过前馈神经网络。
        参数：
        - x: 输入tensor，形状为 (batch_size, seq_len, input_dim)
        """
        # 通过第一个线性层
        x = self.linear1(x)
        # 应用激活函数ReLU
        x = self.relu(x)
        # 通过第二个线性层
        x = self.linear2(x)
        return x
# 示例用法
input_dim = 512  # 输入维度
hidden_dim = 2048  # 隐藏层维度
output_dim = 512  # 输出维度，通常等于输入维度
ffn = FeedForwardNetwork(input_dim, hidden_dim, output_dim)
x = torch.randn(2, 10, input_dim)  # batch_size=2, seq_len=10, input_dim=512
# 通过FFN进行前向传播
output = ffn(x)
print(output.shape)  # 输出形状为 (2, 10, 512)
