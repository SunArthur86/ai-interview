import torch
import torch.nn as nn
import math
import torch.nn.functional as F
from typing import *
class Attention(nn.Module):
    def __init__(self, args):
        # 此处省略了和 KV cache 无关的初始化代码
        self.cache_k = torch.zeros(
            (args.max_batch_size, args.max_seq_len, self.n_local_heads, self.head_dim)
        ).cuda()
        self.cache_v = torch.zeros(
            (args.max_batch_size, args.max_seq_len, self.n_local_heads, self.head_dim)
        ).cuda()
    def forward(self, x: torch.Tensor, start_pos: int, freqs_cis: torch.Tensor, mask: 
Optional[torch.Tensor]):
        bsz, seqlen, _ = x.shape
        xq, xk, xv = self.wq(x), self.wk(x), self.wv(x)
        xq = xq.view(bsz, seqlen, self.n_local_heads, self.head_dim)
        xk = xk.view(bsz, seqlen, self.n_local_heads, self.head_dim)
        xv = xv.view(bsz, seqlen, self.n_local_heads, self.head_dim)
        xq, xk = apply_rotary_emb(xq, xk, freqs_cis=freqs_cis)
        self.cache_k = self.cache_k.to(xq)
        self.cache_v = self.cache_v.to(xq)
        self.cache_k[:bsz, start_pos : start_pos + seqlen] = xk
        self.cache_v[:bsz, start_pos : start_pos + seqlen] = xv
        keys = self.cache_k[:bsz, : start_pos + seqlen]
        values = self.cache_v[:bsz, : start_pos + seqlen]
        xq = xq.transpose(1, 2)
        keys = keys.transpose(1, 2)
        values = values.transpose(1, 2)
        scores = torch.matmul(xq, keys.transpose(2, 3)) / math.sqrt(self.head_dim)
        if mask is not None:
            scores = scores + mask  # (bs, n_local_heads, slen, cache_len + slen)
        scores = F.softmax(scores.float(), dim=-1).type_as(xq)
        output = torch.matmul(scores, values)  # (bs, n_local_heads, slen, head_dim)
        output = output.transpose(
            1, 2
        ).contiguous().view(bsz, seqlen, -1)
        return self.wo(output)

def linear_regression(X, y):
    ones = np.ones((X.shape[0], 1))
    X_b = np.concatenate([ones, X], axis=1)  # 添加x0=1的列
    theta_best = np.linalg.inv(X_b.T.dot(X_b)).dot(X_b.T).dot(y)
    return theta_best
theta_best = linear_regression(X, y)
print("最佳拟合参数:", theta_best)
# 第二种，使用SGD
def compute_cost(X, y, theta):
    m = len(y)
    predictions = X.dot(theta)
    cost = (1 / (2 * m)) * np.sum(np.square(predictions - y))
    return cost
def gradient_descent(X, y, theta, alpha, num_iters):
    m = len(y)
    J_history = np.zeros(num_iters)
    
    for i in range(num_iters):
        predictions = X.dot(theta)
        error = np.dot(X.transpose(), (predictions - y))
        descent = alpha * (1 / m) * error
        theta -= descent
        J_history[i] = compute_cost(X, y, theta)
        
    return theta, J_history
# 初始化参数

# 设置超参数
alpha = 0.01
num_iters = 1000
# 添加x0=1的列
ones = np.ones((X.shape[0], 1))
X_b = np.concatenate([ones, X], axis=1)
# 梯度下降
theta, _ = gradient_descent(X_b, y, theta, alpha, num_iters)
print("梯度下降后的参数:", theta)

10.26 手撕 K-Means聚类
实现K-Means 聚类，要求包含Fit和Predict两个函数。

代码实现：
import numpy as np
 
class KMeans:
    def __init__(self, k=3, max_iters=100, tol=1e-4):
        self.k = k
        self.max_iters = max_iters
        self.tol = tol
    def fit(self, X):
        # 初始化质心
        indices = np.random.choice(X.shape[0], self.k, replace=False)
        centroids = X[indices]
        for _ in range(self.max_iters):
            # 将每个点分配给最近的质心
            clusters = [[] for _ in range(self.k)]
            for features in X:
                distances = [np.linalg.norm(features - centroid) for centroid in centroids]
                closest_cluster = np.argmin(distances)
                clusters[closest_cluster].append(features)
            # 计算新的质心
            new_centroids = np.array([np.mean(cluster, axis=0) for cluster in clusters if cluster])
            # 检查质心是否变化
            if np.allclose(centroids, new_centroids, atol=self.tol):
                break
            centroids = new_centroids
        self.centroids = centroids
        self.clusters = clusters
    def predict(self, X):
        y_pred = [np.argmin([np.linalg.norm(x - centroid) for centroid in self.centroids]) for x in 
X]
        return np.array(y_pred)
# 示例使用
if __name__ == "__main__":
    # 生成一些随机数据
    from sklearn.datasets import make_blobs
    X, _ = make_blobs(n_samples=300, centers=4, cluster_std=0.60, random_state=0)
 
    # 创建KMeans实例并拟合数据
    kmeans = KMeans(k=4)
    kmeans.fit(X)
 
    # 预测每个点的簇标签
    y_pred = kmeans.predict(X)
 
    # 打印质心
    print("Centroids:")
    print(kmeans.centroids)
 
    # 你可以使用matplotlib来可视化结果
    import matplotlib.pyplot as plt
    plt.scatter(X[:, 0], X[:, 1], c=y_pred, cmap='viridis', marker='o', edgecolor='k')
    plt.scatter(kmeans.centroids[:, 0], kmeans.centroids[:, 1], c='red', s=200, alpha=0.75)
    plt.show()

10.27 SFT的局限？
•
sft无法负反馈

sft只知道 next_token 出什么是正确的，而不知道 next_token 出什么是错误的，sft 一直在通过“孤立”来降低错误 token 的出现
概率。
越是在 sft 阶段告诉它什么是错误的，它越是容易提高错误 token 的概率。
站在模型的角度来思考，这个现象非常合理：“训练者不断让我提高 Prob( E | ABCD ) 的概率，那我举一反三，顺带提高一下 Prob( 
E | ACD ) 的概率是不是也合理？训练者是不是应该表扬我？”。
当B 这个 token，恰好是“not”，恰好是“不”，就会诱导模型产生严重的幻觉。
•
sft不具有向后看的能力
在 sft 的训练过程中，每一个 token 都只看得见前面的 token。还是那个经典例子，“台湾不是中国的，这个观点是严重错误的”。
无论你用什么炼丹技巧来做 sft，Prob(中国 | 台湾不是) 的概率都是在增加的，模型无法利用“后半个句子在否定前半句子”这个重
要信息。
在做rlhf时，reward 反向衰减传递的时候，把最大的奖励赏赐给“错误 ”这个 token，而“中国 ”这个 token 可能并不会得到很多
的 reward。

def loss_func(loss_mask, output_tensor):
    
    losses = output_tensor.float()
    loss_mask = loss_mask.view(-1).float()
    loss = torch.sum(losses.view(-1) * loss_mask) / loss_mask.sum()
    # Reduce loss for logging.

    return loss, {'lm loss': averaged_loss[0]}
def average_losses_across_data_parallel_group(losses):
    """Reduce a tensor of losses across all GPUs."""
    averaged_losses = torch.cat(
        [loss.clone().detach().view(1) for loss in losses])
    torch.distributed.all_reduce(averaged_losses,
                                 group=mpu.get_data_parallel_group())
    averaged_losses = averaged_losses / \
        torch.distributed.get_world_size(group=mpu.get_data_parallel_group())
    return averaged_losses

10.29 SFT阶段怎么做数据packing？
处理变长数据，如果一个micro-batch一条数据，容易造成资源浪费。
使用背包问题的贪心解法作packing
10.30 作packing可能有什么坏处？
1. 削弱了模型对难的短query和短答案的拟合。
a. 在无sft packing得情况下，假设batch_size= 1，那么如果有个短query和短答案在这个batch里，其余补充padding，那么这
个batch的gradient全是这个短文本的gradient，模型对这个query的拟合能力会变强。但是如果SFT packing以后，多个短
文本在一个样本中，这个batch的gradient会被稀释，短文本的拟合就不会特别强。
2. 对多轮对话有影响。
a. 目前没有使用Block Attention的微调框架对于Packing的实现方式往往是在两个conversation之间用Padding Token或者是
EOS Token进行隔断，这种方式无法避免Seq和Seq之间的影响，并且，这种数据格式和多轮对话的数据格式是一致的。
b. 如果在多轮对话中使用EOS Token作为当前轮对话结束的标识，那么我们就会发现，模型仅能依赖BOS Token来作为当前
Seq的标识符（即BOS Token前面的是历史Seq，BOS Token后面的才是模型需要的历史信息），实际上这一点对于现在的
LLM通常来说是难以做到的，因为在没有Block Attention的情况下，Seq和Seq之间的Attention系数会导致模型总是无法避
免的依赖于之前的Seq的内容，因此，如果一个数据集中既出现了多轮对话又使用了Packing的技术，会导致模型无法区分历
史对话和当前对话。如果强行使用Packing技术，那么甚至会存在算力浪费

10.32 以跳过sft阶段直接进行RLHF么？
sft可以类比的是RL中的模仿学习，RLHF就是在此基础上纯跟环境交互的RL。那么和LLM对齐阶段相似的过程有两个，第一是
alpha-go，第二个是自动驾驶。它们两个现实中的人工智能技术，开始的时候都是先做模仿学习，第二步进行跟环境交互的强化学
习。他们都有个共同特点，就是环境复杂，模型如果纯进行RL的话，搜索空间过于庞大，消耗资源较多。尽管alpha-go后续有zeroalpha-go算法，但是建立的基础是reward model是客观的，且不计成本的进行空间搜索。在LLM环境下，由于不存在天然的
reward model，所以建立reward model的人工成本较大，所以不适合不计成本的空间搜索。所以利用sft首先做模仿学习缩小搜索
空间，再利用RLHF进行进一步对齐是必要的。

10.34 为什么RLHF在多轮对话和数学推理上有用？
随着多轮对话的进行，模型解码时每一步推理的attention都会因为序列长度变大而可能被“稀释”，强化学习在每一轮交互中的奖

活多变的交互， 交互粒度越细，good与bad界限越清晰， 强化学习相对sft的提升就越明显。

10.35 RLHF 最近的新算法有哪些，介绍几个？
参考链接：https://zhuanlan.zhihu.com/p/710021282
大厂面试官问RLHF及变体时，如何优雅地装杯
1. 如果数据集是人类标注的，就是RLHF
2. 如果是gpt，PaLM2等模型标注的，就是RLAIF
3. RL技术中，典型的是PPO（缺点是训练不太稳定，效率低，资源消耗大）
4. 使用一种好的方法来计算 baseline 是丢掉 Critic 网络的关键，就是GRPO
5. 我们可以丢掉 Critic（教练），Actor 不再需要受到 Critic 的指导，而是直接去对齐 RM，就是ReMax
6. 后面出现了非RL的技术，典型的就是DPO（直接构造偏好数据集进行优化）
7. 然后有一些工作觉得SFT+RLHF不是end2end，索性直接在SFT上加一个alignment的loss，这个典型的就是ORPO
8. 还有一部分工作觉得DPO训练的pair-wise（x,y_w,yl）的数据集获取很困难，于是有了point-wise(x,y,label)的方法，典型的是
KTO
9. 还有一些就是优化Alignment的输出长度的，加了一个类似正则化项的限制DPOP
10. 最近又出现了一种step-wise的优化方法(例如Step-DPO)，直接对reasoning的过程进行优化限制(step-level)，而不是等模型输
出完了再进行优化(Instance-level)
11. 有人觉得reward model无法优化general perference（不可传递性偏好，比如a>b,b>c, c>a的情况）于是有了DNO，SPO（引
入了纳什均衡，Nash equilibrium）
12. 有的人觉得DPO优化的是单轮，因此对多轮的偏好优化做了拓展（multi-turn DPO）

13. 方法超不过你，就直接合成数据，用MCTS（exploration and exploitation保证多样性和质量）等方式造完数据，直接DPO开
始训练就完了，完美的数据闭环

10.38 LORA 介绍，秩的数学含义，怎么计算矩阵的秩？
•
秩是指矩阵中线性无关的行或列的最大数目
•
具体地说：
◦
线性独立性：矩阵的秩是矩阵中线性独立行或列的数量。如果矩阵的某些行或列可以通过其他行或列的线性组合来表示，那
么这些行或列就不是线性独立的。
◦
维度：矩阵的秩可以告诉我们矩阵的列空间或行空间的维度。秩越高，矩阵的列空间或行空间的维度就越大。
◦
解的个数：在线性方程组中，矩阵的秩可以决定方程组解的个数。如果增广矩阵的秩等于系数矩阵的秩，并且等于变量的个
数，那么方程组有唯一解；如果增广矩阵的秩小于系数矩阵的秩，那么方程组无解；如果增广矩阵的秩小于变量的个数，那
么方程组有无穷多解。
◦
矩阵的逆：如果矩阵是方阵，那么只有当矩阵的秩等于其阶数时，矩阵才是可逆的。
•
计算方式
◦
通过初等行变换将矩阵转换为阶梯形或行最简形，然后计算非零行的数量

10.44 讲讲你了解的scaling law有哪些？
🌟目前关于LLM的Scaling Law研究主要集中在模型性能与规模（参数、数据、计算量）之间的关系、涌现能力的发现、推理
优化以及实际应用中的挑战等方面。
基础Scaling Law研究 [1]
•
Kaplan等人的经典理论：2020年提出的Scaling Law指出，模型性能（如测试损失）随参数数量、数据集大小和计算量的指数增
长呈线性提升，且三者需协同扩展以避免瓶颈。例如，模型参数增加约8倍时，训练数据需增加约5倍以维持性能提升。
•
幂律关系的验证：研究发现，测试损失与模型规模之间遵循逆幂律关系，即损失随规模增大而平滑下降，但提升速度逐渐放缓。
这一现象在多个数据集上被复现，成为预训练扩展的理论基础。
强化Scaling Law研究 [2]
•
过优化Goodhart’s law：模型可能过度拟合代理奖励函数，导致在真实评估指标上的性能下降，即Reward Hacking。这符合
Goodhart定律——“当指标成为目标时，它就不再是一个好指标”。论文通过合成实验验证了这一现象：随着模型在代理奖励上
的得分提升，其在“黄金标准”奖励模型（代表真实偏好）上的得分先上升后下降，呈现“驼峰型”曲线。
•
直接对齐算法的隐式过度优化：
◦
KL散度与性能的非线性关系：不同KL约束下，模型性能（如GPT-4评估的胜率）随训练步长增加先升后降，且小模型更易陷
入过度优化。
◦
模型规模的影响：大模型在KL与性能的权衡中表现更优，表明模型容量越高，对隐式奖励的泛化能力越强。
微调Scaling Law研究 [3]
•
幂律联合缩放法则：
◦
微调性能遵循幂律乘法联合缩放法则，即模型性能与微调数据量、模型大小、预训练数据量等因素呈幂律关系。例如，测试
PPL随模型参数和微调数据量的增加而平滑下降，但提升速度逐渐放缓。
◦
当模型大小（如从 1B 到 16B）与微调数据量协同扩展时，性能提升显著，但参数高效调优（PET）方法的参数量扩展对性能
提升作用有限
•
模型规模 > 预训练数据规模：模型大小的扩展比预训练数据量的扩展更有效，且发现预训练数据量的影响仅在模型规模较小时显
著，而对大模型的微调增益有限。
•
PEFT的适用性：
◦
PEFT在数据量较少时表现更优，而全模型调优（FMT）需要更多数据但最终效果更好13。
◦
LoRA 的稳定性优于 Prompt Tuning：LoRA 在数据量增加时表现更稳定，而 Prompt Tuning 可能出现逆向扩展（性能随参
数增加而下降）。
◦
PEFT 方法的泛化能力更强，因其冻结了大部分预训练参数，保留基础模型的知识。
•
任务与数据的依赖：最优微调方法高度依赖任务类型和微调数据量。例如：
◦
数据量少（几千样本）：优先选择 PET（如 LoRA 或 Prompt Tuning）。
◦
数据量中等（数万样本）：LoRA 更稳定。
◦
数据量极大（百万级）：FMT 效果更优。
推理Scaling Law研究 [4]

•
慢思考策略：通过额外计算资源（如多次采样、树搜索）提升推理质量，例如宽度扩展（BoN、MCTS）和生成-选择机制。研究
发现，推理路径长度增加会导致信息损失（雪球误差效应），需平衡生成与选择步骤的效率和准确性。
•
推理成本与效率：随着模型规模扩大，推理成本急剧上升，行业开始转向优化推理阶段而非单纯扩展预训练。
批大小Scaling Law研究 [5]
•
激增现象：Adam类优化器的最优学习率随Batch Size的增加呈现先上升后下降的非单调趋势，类似“浪潮”形态。例如，当
Batch Size较小时，增大Batch Size可提升学习率；但超过临界值（由噪声水平 Bnoise决定）后，学习率反而需降低以维持收敛
稳定性。
•
与SGD优化器的对比：
◦
SGD的最优学习率随Batch Size线性或平方根缩放（如 η∝B或 η∝B），主要基于梯度噪声方差的分析。
◦
Adam的激增现象源于其自适应学习率机制和符号更新策略，导致其优化路径对Batch Size的敏感度更高。
•
训练过程中的动态变化：随着训练进行，BnoiseBnoise 逐渐增大，导致激增现象的峰值向更大的Batch Size方向移动。这表明
不同训练阶段需动态调整学习率与Batch Size的配比。

10.47 对于大模型复杂推理任务，有哪些主流方法
💡复杂推理任务是LLM目前讨论的最多的基础能力，试图通过增强LLM的复杂推理能力来幻觉LLM的幻觉问题，同时赋予LLM
更多的可能性。
一、 Prompt Engineering

一句话总结：仅仅通过prompt最大化地激发出LLM的知识，从而获得更好的推理结果。
1. COT
Chain-of-Thought（CoT） 是一种让LLM通过分步推理生成答案的技术。其核心是模仿人类思考过程，将复杂问题拆解为多个中间
步骤，逐步推导出最终答案，而非直接输出结果。
在实现上，可以分为zero shot（let's step by step）和few shot（给予多个无偏示例，展示如何分步解决问题）。
•
示例
💡问题：28名学生需要乘车，每辆车坐6人，至少需要几辆车？
CoT回答：28 ÷ 6 = 4余4 → 需要4辆车坐满，剩余4人需1辆车 → 共5辆。
•
优缺点
优点：
•
降低单步错误率，增强答案可靠性。
•
推理过程透明化，便于验证逻辑是否合理。
•
CoT生成的推理链可作为高质量数据，用于微调模型。
缺点：
•
模型需要逐句输出推理过程，消耗更多计算资源和时间，影
响实时性。
•
少样本CoT依赖提供的示例质量，若示例存在偏差或错误，
模型可能模仿错误模式。

•
对无需复杂推理的问题（如事实问答），分步解释反而降低
效率。
2. ToT
Tree of Thought（ToT） 是一种更复杂的推理框架，通过多路径探索和动态决策提升LLM解决复杂问题的能力。其核心是将单一线
性CoT扩展为树状结构，允许模型在推理过程中生成多个可能的路径，并基于反馈选择最优路径。
•
关键实现方式
💡节点生成：将问题分解为多个中间步骤（节点），每个节点代表一种可能的推理方向。
评估与剪枝：对每个节点进行评分（如正确性、效率），淘汰低质量分支以节省资源。
深度优先/广度优先搜索：按需选择搜索策略，平衡计算成本与答案质量。
3. SC
Self-Consistency（SC） 是一种基于CoT的增强推理方法，旨在通过生成多条推理路径并选择最一致的答案，提升LLM在复杂任务
中的准确性和鲁棒性。其核心思想是：单一推理路径可能因随机性出错，但多条独立路径的统计共识更可能接近正确答案。
•
核心实现
💡生成多样性：通过调整Temperature或随机种子，让模型对同一问题生成多条不同的推理路径（即不同的“思维链”）。
答案聚合：从所有生成的路径中提取最终答案，通过多数投票（Majority Voting） 或 加权评分 选择出现频率最高的答
案。
•
实现细节
💡路径数量：通常取5~40条，需权衡成本与精度（N越大效果越好，但成本线性增长）。
温度参数：适当调高温度（如0.7~1.0）以增加生成多样性，但过高可能导致逻辑混乱。
答案对齐：需设计规则处理格式差异（如“五辆”和“5”需归一化为同一答案）。
4. PAL
Program-Aided Language Models (PAL) 是一种通过生成可执行程序代码来解决复杂问题的LLM增强方法。其核心思想是让模型
将自然语言问题转化为代码（如Python），并通过执行代码获得精确答案，而非依赖模型直接计算或推理。这种方法尤其擅长数
学、逻辑和符号推理任务，结合了语言模型的灵活性与代码执行的准确性。
•
核心实现
💡代码即中间表示：模型将问题拆解为可编程逻辑，生成代码片段（如计算、循环、条件判断）。
外部执行验证：生成的代码由外部解释器（如Python解释器）执行，确保计算结果的数学正确性。
二、 Training
1. SFT
通过在特定的推理数据集上进行SFT，能让模型更精确地掌握某类推理任务。
常用的推理数据集包括：
💡•
GSM8K：中小学数学题集。
•
MATH：涵盖高中及竞赛数学题，考察较高难度的数学推理。
•
LogiQA：考察抽象的逻辑推理与阅读理解能力。

核心流程
💡1. 使用语言模型生成回答；
2. 人类标注者对回答进行评分或排序；
3. 训练一个Reward Model来模拟人类对回答优劣的判断；
4. 用 PPO 等强化学习算法，让模型在每次生成回答后，根据「奖励」进行参数更新，倾向输出更符合人类需求和逻辑的
内容。
3. Critic Models
有些研究会在主模型之外，再训练一个Critic Models来检查主模型的推理步骤，一旦发现不合理之处，就提出警示或要求修正，进
而将这类评价结果反馈到训练或推断过程中，实现对主模型推理质量的改进。
•
核心思路
💡有一个主模型（或者说「生成模型」）进行推理、回答。
有一个次级模型（或者说「批判模型」或「验证器」）对主模型的理由链、过程正确性或最终输出进行打分或判断。
以此打分或判断为依据来改进主模型：比如过滤掉错误推理，或通过强化学习的方式更新主模型参数等。
近期常见的 PRM/ORM 或类似的多级 reward 机制，通常就是「自动验证器与批判模型」在实践中的一个具体实现 。它们
都为主模型生成的过程或结果提供一种「外部批判/评价」，从而帮助主模型提升推理一致性、正确性或安全性。
•
PRM（Process Reward Model）
PRM 的目标是给「推理过程」给一个reward，比如给每一步的思路链、Chain-of-Thought 提供一个质量评价。
这意味着它更关注推理展开的细节：是否逻辑连贯、中间步骤有没有硬伤等。
这种过程级评估非常符合「批判模型」或「验证器」的角色——它不仅看最终答案对不对，更会审查一系列中间步骤的合理性。
•
ORM（Output Reward Model）
ORM 则主要打分「最终输出」的质量。
与 PRM 相比，ORM 只看结果是否正确、合乎人类偏好或任务标准，而不深入逐步过程。
这种单一关注输出的评估模式，也可视为一种更「精简」的批判模型——它仍然在对主模型的推理结论进行审核。
另外，若结合外部符号推理器进行正式验证，还可以将数学推理或逻辑证明转化为可机读的公式化、形式化的形式，进而用定理证
明器进行检验，保证结论正确。不过，目前公式化、形式化的验证仅能涵盖少数对逻辑一致性要求极高的任务，对日常复杂语言仍
有难度。
