# 二分类交叉熵loss
def binary_cross_entropy_loss(y_pred, y_true):
# 应用 Sigmoid 将模型输出转换为概率
y_pred = 1 / (1 + torch.exp(-y_pred))

# 防止 log(0) 导致的数值不稳定,使用 epsilon 做平滑处理
epsilon = 1e-7
# 限制预测值在 (epsilon, 1-epsilon) 范围内
y_pred = torch.clamp(y_pred, min=epsilon, max=1.0 - epsilon)
# 计算交叉熵损失
loss = - (y_true * torch.log(y_pred) + (1 - y_true) * torch.log(1 - y_pred))
return loss.mean()

import numpy as np
# 定义sigmoid和其导数
def sigmoid(z):
return 1 / (1 + np.exp(-z))
def sigmoid_derivative(z):
return sigmoid(z) * (1 - sigmoid(z))
# 定义model以及训练推理代码
class SimpleNN:
def __init__(self, input_size, hidden_size, output_size):
# 初始化权重和偏置
self.weights1 = np.random.randn(input_size, hidden_size)
self.bias1 = np.zeros((1, hidden_size))
self.weights2 = np.random.randn(hidden_size, output_size)
self.bias2 = np.zeros((1, output_size))
def forward(self, X):
self.Z1 = np.dot(X, self.weights1) + self.bias1
self.A1 = sigmoid(self.Z1)
self.Z2 = np.dot(self.A1, self.weights2) + self.bias2
self.A2 = sigmoid(self.Z2)
return self.A2
def backward(self, X, y, learning_rate):
# 样本数量
m = X.shape[0]
# 计算输出层的误差
dA2 = -y / self.A2 + (1 - y) / (1 - self.A2)
dZ2 = dA2 * sigmoid_derivative(self.Z2) # dZ2其实可以直接等于 self.A2-y
dW2 = np.dot(self.A1.T, dZ2) / m
db2 = np.sum(dZ2, axis=0, keepdims=True) / m
# 计算隐藏层的误差
dA1 = np.dot(dZ2, self.weights2.T)
dZ1 = dA1 * sigmoid_derivative(self.Z1)
dW1 = np.dot(X.T, dZ1) / m
db1 = np.sum(dZ1, axis=0, keepdims=True) / m
# 更新权重和偏置
self.weights1 -= learning_rate * dW1
self.bias1 -= learning_rate * db1
self.weights2 -= learning_rate * dW2
self.bias2 -= learning_rate * db2
def compute_loss(self, X, y):
# 计算交叉熵损失
A2 = self.forward(X) m = X.shape[0]
loss = - np.mean(y * np.log(A2) + (1 - y) * np.log(1 - A2))
return loss
def train(self, X_train, y_train, epochs, learning_rate):
# 训练模型
for epoch in range(epochs):
self.forward(X_train) # 前向传播
self.backward(X_train, y_train, learning_rate) # 反向传播
# 每100次迭代打印一次损失
if epoch % 100 == 0:
loss = self.compute_loss(X_train, y_train)
print(f"Epoch {epoch}/{epochs}, Loss: {loss:.4f}")
def predict(self, X):

# 推理,返回类别标签
A2 = self.forward(X)
return (A2 > 0.5).astype(int)
